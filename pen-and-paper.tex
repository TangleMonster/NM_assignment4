    \section*{Question 1}
    First let
    \[
        x_j=-1+jh,\qquad h=\frac{2}{n+1},\qquad j=0,1,\dots,n+1,
    \]
    and let $u\in C^4([-1,1])$ with $u(-1)=u(1)=0$, defining
    \[
        V_j:=\frac{u(x_{j-1})-2u(x_j)+u(x_{j+1})}{h^2}, \qquad j=1,\dots,n
    \]
    Then using Taylor expansion around xj,
    \begin{align*}
        u(x_{j+1}) &= u(x_j)+h u'(x_j)+\frac{h^2}{2}u''(x_j)+\frac{h^3}{6}u^{(3)}(x_j)+\frac{h^4}{24}u^{(4)}(\xi_j^+),\\
        u(x_{j-1}) &= u(x_j)-h u'(x_j)+\frac{h^2}{2}u''(x_j)-\frac{h^3}{6}u^{(3)}(x_j)+\frac{h^4}{24}u^{(4)}(\xi_j^-),
    \end{align*}
    for some $\xi_j^\pm$ between $x_{j-1}$ and $x_{j+1}$.\\
    Then adding both lines and subtracting $2u(x_j)$
    \[
        u(x_{j-1})-2u(x_j)+u(x_{j+1})=h^2u''(x_j)+\frac{h^4}{24}\big(u^{(4)}(\xi_j^-)+u^{(4)}(\xi_j^+)\big).
    \]
    Hence
    \[
        V_j-u''(x_j)=\frac{h^2}{24}\big(u^{(4)}(\xi_j^-)+u^{(4)}(\xi_j^+)\big).
    \]
    Since $u^{(4)}$ is continuous, so there exist $M>0$ such that
    $|u^{(4)}(x)|\le M$ for all $x\in[-1,1]$. Therefore
    \[
        |V_j-u''(x_j)|\le \frac{h^2}{24}(M+M)=\frac{M}{12}h^2.
    \]
    Let $K=\frac{M}{12}$, then take the maximum over interior indices:
    \[
        \max_{1\le j\le n}|V_j-u''(x_j)|\le Kh^2.
    \]
    So the differentiation matrix is second-order accurate in $\|\cdot\|_\infty$.

    First Crank-Nicholson scheme:
    \[
        u_{i+1} = u_i + \frac{\tau}{2}\left[f(t_{i+1}, u_{i+1}) + f(t_i, u_i)\right]
    \]
    then substitute $f(t,y) = Ay + g(t)$, that
    \[
        u_{i+1} = u_i + \frac{\tau}{2}\left[Au_{i+1} + g(t_{i+1}) + Au_i + g(t_i)\right]
    \]
    then rearrange the terms, put $u_{i+1}$ on left and $u_i$ on the right
    \begin{align*}
        u_{i+1} = u_i + \frac{\tau}{2}Au_{i+1} + \frac{\tau}{2}g(t_{i+1}) + \frac{\tau}{2}Au_i + \frac{\tau}{2}g(t_i) \\
        => \qquad u_{i+1} - \frac{\tau}{2}Au_{i+1} = u_i + \frac{\tau}{2}Au_i + \frac{\tau}{2}g(t_{i+1})  + \frac{\tau}{2}g(t_i) \\
        => \qquad (I-\frac{\tau}{2}A) u_{i+1} = (I+\frac{\tau}{2}A) u_i + \frac{\tau}{2}g(t_{i+1})  + \frac{\tau}{2}g(t_i)
    \end{align*}
    so we can extract:
    \begin{align}
        B = I - \frac{\tau}{2}A \\
        L = I + \frac{\tau}{2}A \\
        f_{i} = \frac{\tau}{2}[g{t_{i+1}} + g(t_{i})]
    \end{align}
    Thus, with the initial condition is $u_0 = y_0$, the Crank-Niocholson schmem can be written as:
    \[
        Bu_{i+1} = Lu_{i} + f_{i}
    \]

    \section*{Question 9:}
    First Forward Euler: $U_{k+1} = U_{k} + {\tau}DU_{k})$ => $(I + {\tau}D)U_{k}$, so the iteration martix is $M = I + {\tau}D$.
    If Forward Euler is stable, all absolute value of eigenvalue in martix M must less than 1. So the eigenvalue $\mu_j$ that $\mu_j = 1 + \tau \lambda_j$ satisfied
    \begin{align*}
        |\mu_j| < 1 \\
        => \qquad |1 + \tau \lambda_j| < 1 \\
        => \qquad -1 < 1 + \tau \lambda_j < 1\\
        => \qquad -2 < \tau \lambda_j < 0
    \end{align*}
    Since
    \[
        \lambda_j = -\frac{4}{h^2}\sin^2\left(\frac{j\pi}{2(n+1)}\right)
    \]
    that $\lambda_j$ is always negative, so $\tau \lambda_j < 0$ always satified.\\
    Then we get
    \[
       \tau < -\frac{2}{\lambda_j}
    \]
    Next calcu the min value of the right side:
    \[
        \tau < \min_j\left( -\frac{2}{\lambda_j}\right)
    \]
    \[
        \qquad => \qquad \max_j |\lambda_j| = \max_j \left|-\frac{4}{h^2}\sin^2\left(\frac{j\pi}{2(n+1)}\right)\right| = \max_j \left[\frac{4}{h^2}\sin^2\left(\frac{j\pi}{2(n+1)}\right)\right]
    \]
    It meet the max when $j=n$, and $\max_n\left(\sin^2\left(\frac{n\pi}{2(n+1)}\right)\right) = 1$\\
    Thus,
    \[
        \max_j |\lambda_j| = \frac{4}{h^2} \qquad=>\qquad  \tau < -\frac{2}{-\frac{4}{h^2}} \qquad=>\qquad \tau < \frac{h^2}{2}
    \]
    In conclusion, the Forward Euler scheme for the heat conduction equation is conditionally stable, provided that the time step satisfies $\tau < \frac{h^2}{2}$


    \section*{Question 10:}
    First we got
     \[
        B = I - \frac{\tau}{2}A, \quad L = I + \frac{\tau}{2}A, \quad f_{i} = \frac{\tau}{2}[g{t_{i+1}} + g(t_{i})],
     \]
    from question 7, and in this question we have
    \[
        U'(t) = DU(t), \quad t \geq 0, \quad U(0) = \varphi,
    \]
    so that
    \[
        BU_{k+1} = LU_k, \quad B = I - \frac{\tau}{2}D, \quad L = I + \frac{\tau}{2}D,
    \]
    Since D and B has the same eigenvectors, the eigenvalues of B are:
    \[
        \beta_j = 1-\frac{\tau}{2}\lambda_j,
    \]
    Because $\lambda_j < 0$, $\beta_j > 1$ ($\beta_j \not= 0$ for all j) , so B is invertible. Then the linear system $BU_{k+1} = LU_k$ has unique solution at each step.
    *Hence the Crank Nicholson scheme for the discretised heat equation is well-defined*.\\


    2) The scheme is unconditionally asymptotically stable. \\


    First the iteration matrix for the scheme is $M = B^{-1}L$. Using the given result, if $(\beta_{j}, z_{j})$ is an eigenpair for B and $(\gamma_j, z_j)$ an eigenpair of L (To avoid confusion, I use $\gamma$ to replace $\lambda$ in the question), then $(\beta^{-1}\gamma_{j},z_j)$ is an eigenpair of M.\\
    The eigenvalues of L are:
    \[
        \gamma_j = 1 + \frac{\tau}{2}\lambda_j,
    \]
    therefore, the eigenvalues of M are:
    \[
        \mu_j = \frac{\gamma_j}{\beta_j} = \frac{1+\frac{\tau}{2}\lambda_j}{1-\frac{\tau}{2}\lambda_j}
    \]
    Second we need to proof that $|\mu_j| < 1$ for all $\tau > 0$:\\
    Since $\lambda_j < 0$, we got
    \[
        \begin{gathered}
            \frac{\tau}{2}\lambda_j < 0 \\
            => 1 + \frac{\tau}{2}\lambda_j < 1 \\
            => 1 - \frac{\tau}{2}\lambda_j > 1 \\
            Thus \quad \mu_j = \frac{1+\frac{\tau}{2}\lambda_j}{1-\frac{\tau}{2}\lambda_j} < 1
        \end{gathered}
    \]
    Then if $\mu_j > -1$:
    \[
        \begin{gathered}
            \frac{1+\frac{\tau}{2}\lambda_j}{1-\frac{\tau}{2}\lambda_j} > -1 \\
            => 1 + \frac{\tau}{2}\lambda_j > -(1 - \frac{\tau}{2}\lambda_j) \\
            => 1 + \frac{\tau}{2}\lambda_j > \frac{\tau}{2}\lambda_j-1 \\
            => 1 > -1
        \end{gathered}
    \]
    Therefore, $\mu_j > -1$ always true.
    Hence, $|\mu_j| < 1$ for all $\tau > 0$ that the scheme is unconditionally asymptotically stable.
